### 3. Build Your First Custom Agent

<Steps>

    Navigate to [Advanced Scenario Creator](https://app.toughtongueai.com/advanced-scenario-create)
  </Step>

  <Step title="Describe What You Want">

  </Step>

    Copy the link and share, or embed in your training portal.
  </Step>
</Steps>

### 4. Handle Iframe Events

Listen to lifecycle events from the embedded iframe:

```javascript
window.addEventListener("message", (event) => {
  const data = event.data;

  switch (data.event) {
    case "onStart":
      console.log("Session started:", data.sessionId);
      // Track in your analytics
      break;

    case "onStop":
      console.log("Session completed:", data.sessionId);
      // Fetch analysis via API
      break;

    case "onReady":
      console.log("Iframe ready");
      break;
  }
});
```

### 5. Example: End-to-End Flow

Here's a complete example using Next.js:

<CodeGroup>
```typescript API Route (app/api/analyze/route.ts)
export async function POST(req: Request) {
  const { session_id } = await req.json();
  const apiKey = process.env.TOUGH_TONGUE_API_KEY;

const response = await fetch(
'https://api.toughtongueai.com/api/public/sessions/analyze',
{
method: 'POST',
headers: {
'Authorization': `Bearer ${apiKey}`,
'Content-Type': 'application/json',
},
body: JSON.stringify({ session_id }),
}
);

return NextResponse.json(await response.json());
}

````

```typescript Frontend (app/practice/page.tsx)
'use client';

export default function PracticePage() {
  useEffect(() => {
    const handleMessage = async (event: MessageEvent) => {
      if (event.data.event === 'onStop') {
        const response = await fetch('/api/analyze', {
          method: 'POST',
          body: JSON.stringify({ session_id: event.data.sessionId }),
        });
        const analysis = await response.json();
        console.log('Analysis:', analysis);
      }
    };

    window.addEventListener('message', handleMessage);
    return () => window.removeEventListener('message', handleMessage);
  }, []);

  return (
    <iframe
      src="https://app.toughtongueai.com/embed/YOUR_SCENARIO_ID"
      width="100%"
      height="600px"
      allow="microphone"
    />
  );
}
````
